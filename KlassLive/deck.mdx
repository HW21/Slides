## Redux 

```javascript
const updateProblemTitle = (klass, action) => {
    const {problemId, title} = action;
    if (!problemId) return klass;

    const problem = klass.problemsById[problemId];
    if (!problem) return klass;

    return {
        ...klass,
        problemsById: {
            ...klass.problemsById,
            [problemId]: {
                ...problem,
                title: title
            }
        }
    }
};
```
